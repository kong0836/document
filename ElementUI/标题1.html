# 常见问题

# 表单校验时焦点自动跳转到第一个校验失败的位置

this.$refs[formName].validate((valid) => {
    if (!valid) {
      setTimeout(()=>{
        var isError= document.getElementsByClassName("is-error");
        isError[0].querySelector('input').focus();
      },100);
      return false;
    }
});

# Collapse 折叠面板默认全部展开

<template>
  <el-collapse v-model="activeName" v-for="(item,index) in demoList" :key="index">
    <el-collapse-item :title="item.name" :name="item.id">
    </el-collapse-item>
  </el-collapse>
</template>

<script>
export default {
  props: {
    demoList: {
      type: Array,
      default: null
    }
  }
  computed:{
    activeName() {
      // demo.id 数据唯一标识
      return this.demoList.map((demo) => demo.id);
    }
  }
}
</script>

# 布局容器纵向高度自适应

  • 在 ``index.html或者App.vue文件内指定以下css` 样式

    html, body, #app {
  height: 100%;
  margin: 0;
  padding: 0;
}

  • 指定 el-container 容器的高度

      <el-container style="height:100%;"></el-container>
